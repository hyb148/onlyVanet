78 void VANETMobilityBase::initialize(int stage)
79 {
80  cSimpleModule::initialize(stage);
81  EV_TRACE << "initializing VANETMobilityBase stage " << stage << endl;
82  if (stage == 0)
83  {
84  constraintAreaMin.x = par("constraintAreaMinX");
85  constraintAreaMin.y = par("constraintAreaMinY");
86  constraintAreaMin.z = par("constraintAreaMinZ");
87  constraintAreaMax.x = par("constraintAreaMaxX");
88  constraintAreaMax.y = par("constraintAreaMaxY");
89  constraintAreaMax.z = par("constraintAreaMaxZ");
90  visualRepresentation = findVisualRepresentation();
91  if (visualRepresentation) {
92  const char *s = visualRepresentation->getDisplayString().getTagArg("p", 2);
93  if (s && *s)
94  error("The coordinates of '%s' are invalid. Please remove automatic arrangement"
95  " (3rd argument of 'p' tag) from '@display' attribute.", visualRepresentation->getFullPath().c_str());
96  }
97  }
98  else if (stage == 1)
99  {
100  initializePosition();
101  if (!isFiniteNumber(lastPosition.x) || !isFiniteNumber(lastPosition.y) || !isFiniteNumber(lastPosition.z))
102  throw cRuntimeError("Mobility position is not a finite number after initialize (x=%g,y=%g,z=%g)", lastPosition.x, lastPosition.y, lastPosition.z);
103  if (isOutside())
104  throw cRuntimeError("Mobility position (x=%g,y=%g,z=%g) is outside the constraint area (%g,%g,%g - %g,%g,%g)",
105  lastPosition.x, lastPosition.y, lastPosition.z,
106  constraintAreaMin.x, constraintAreaMin.y, constraintAreaMin.z,
107  constraintAreaMax.x, constraintAreaMax.y, constraintAreaMax.z);
108  EV << "initial position. x = " << lastPosition.x << " y = " << lastPosition.y << " z = " << lastPosition.z << endl;
109  emitMobilityStateChangedSignal();
110  updateVisualRepresentation();
111 
112  if (!areaInitalized)
113  {
114  mininumArea=constraintAreaMin;
115  maximumArea = constraintAreaMax;
116  updateDisplayString();
117  areaInitalized=true;
118  }
119  else
120  {
121 
122  bool change=false;
123  if (isFiniteNumber(mininumArea.x) && mininumArea.x>constraintAreaMin.x)
124  {
125  mininumArea.x=constraintAreaMin.x;
126  change=true;
127  }
128  if (isFiniteNumber(mininumArea.y) && mininumArea.y>constraintAreaMin.y)
129  {
130  mininumArea.y=constraintAreaMin.y;
131  change=true;
132  }
133  if (isFiniteNumber(mininumArea.z) && mininumArea.z>constraintAreaMin.z)
134  {
135  mininumArea.z=constraintAreaMin.z;
136  change=true;
137  }
138  if (isFiniteNumber(maximumArea.x) && maximumArea.x<constraintAreaMax.x)
139  {
140  maximumArea.x=constraintAreaMax.x;
141  change=true;
142  }
143  if (isFiniteNumber(maximumArea.y) && maximumArea.y<constraintAreaMax.y)
144  {
145  maximumArea.y=constraintAreaMax.y;
146  change=true;
147  }
148  if (isFiniteNumber(maximumArea.z) && maximumArea.z<constraintAreaMax.z)
149  {
150  maximumArea.z=constraintAreaMax.z;
151  change=true;
152  }
153  if (change)
154  updateDisplayString();
155  // vanetMobilityStateChangedSignal = registerSignal("vanetMobilityStateChanged");
156  }
157 
158  }

169 void VANETMobilityBase::setInitialPosition()
170 {
171  // reading the coordinates from omnetpp.ini makes predefined scenarios a lot easier
172  bool filled = false;
173  if (hasPar("initFromDisplayString") && par("initFromDisplayString").boolValue() && visualRepresentation)
174  {
175  filled = parseIntTo(visualRepresentation->getDisplayString().getTagArg("p", 0), lastPosition.x)
176  && parseIntTo(visualRepresentation->getDisplayString().getTagArg("p", 1), lastPosition.y);
177  if (filled)
178  lastPosition.z = 0;
179 
180  }
181  // not all mobility models have "initialX", "initialY" and "initialZ" parameters
182  else if (hasPar("initialX") && hasPar("initialY") && hasPar("initialZ"))
183  {
184  lastPosition.x = par("initialX");
185  lastPosition.y = par("initialY");
186  lastPosition.z = par("initialZ");
187  filled = true;
188  }
189  if (!filled)
190  lastPosition = getRandomPosition();
191 }

244 static int reflect(double min, double max, double &coordinate, double &speed)
245 {
246  double size = max - min;
247  double value = coordinate - min;
248  int sign = 1 - FWMath::modulo(floor(value / size), 2) * 2;
249  ASSERT(sign == 1 || sign == -1);
250  coordinate = FWMath::modulo(sign * value, size) + min;
251  speed = sign * speed;
252  return sign;
253 }

255 void VANETMobilityBase::reflectIfOutside(Coord& targetPosition, Coord& speed, double& angle)
256 {
257  int sign;
258  double dummy;
259  if (lastPosition.x < constraintAreaMin.x || constraintAreaMax.x < lastPosition.x) {
260  sign = reflect(constraintAreaMin.x, constraintAreaMax.x, lastPosition.x, speed.x);
261  reflect(constraintAreaMin.x, constraintAreaMax.x, targetPosition.x, dummy);
262  angle = 90 + sign * (angle - 90);
263  }
264  if (lastPosition.y < constraintAreaMin.y || constraintAreaMax.y < lastPosition.y) {
265  sign = reflect(constraintAreaMin.y, constraintAreaMax.y, lastPosition.y, speed.y);
266  reflect(constraintAreaMin.y, constraintAreaMax.y, targetPosition.y, dummy);
267  angle = sign * angle;
268  }
269  if (lastPosition.z < constraintAreaMin.z || constraintAreaMax.z < lastPosition.z) {
270  sign = reflect(constraintAreaMin.z, constraintAreaMax.z, lastPosition.z, speed.z);
271  reflect(constraintAreaMin.z, constraintAreaMax.z, targetPosition.z, dummy);
272  // NOTE: angle is not affected
273  }
274 }

318 void VANETMobilityBase::handleIfOutside(BorderPolicy policy, Coord& targetPosition, Coord& speed, double& angle)
319 {
320  switch (policy)
321  {
322  case REFLECT: reflectIfOutside(targetPosition, speed, angle); break;
323  case WRAP: wrapIfOutside(targetPosition); break;
324  case PLACERANDOMLY: placeRandomlyIfOutside(targetPosition); break;
325  case RAISEERROR: raiseErrorIfOutside(); break;
326  default: throw cRuntimeError("Invalid outside policy=%d in module", policy, getFullPath().c_str());
327  }
