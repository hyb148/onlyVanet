172  template<typename T> T read() {
173  T buf_to_return;
174  unsigned char *p_buf_to_return = reinterpret_cast<unsigned char*>(&buf_to_return);
175 
176  if (isBigEndian()) {
177  for (size_t i=0; i<sizeof(buf_to_return); ++i) {
178  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
179  p_buf_to_return[i] = buf[buf_index++];
180  }
181  } else {
182  for (size_t i=0; i<sizeof(buf_to_return); ++i) {
183  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
184  p_buf_to_return[sizeof(buf_to_return)-1-i] = buf[buf_index++];
185  }
186  }
187 
188  return buf_to_return;
189  }
190 
191  template<typename T> void write(T inv) {
192  unsigned char *p_buf_to_send = reinterpret_cast<unsigned char*>(&inv);
193 
194  if (isBigEndian()) {
195  for (size_t i=0; i<sizeof(inv); ++i) {
196  buf += p_buf_to_send[i];
197  }
198  } else {
199  for (size_t i=0; i<sizeof(inv); ++i) {
200  buf += p_buf_to_send[sizeof(inv)-1-i];
201  }
202  }
203  }
