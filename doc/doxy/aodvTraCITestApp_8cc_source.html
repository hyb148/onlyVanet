31 void aodvTraCITestApp::initialize(int stage)
32 {
33  cSimpleModule::initialize(stage);
34 
35  if (stage == 0)
36  {
37  testNumber = par("testNumber");
38 
39  cModule* host = getContainingNode(this);
40  traci = check_and_cast<aodvTraCIMobility *>(host->getModuleByPath(".mobility"));
41  traci->subscribe(mobilityStateChangedSignal, this);
42 
43  visitedEdges.clear();

45

55  }
