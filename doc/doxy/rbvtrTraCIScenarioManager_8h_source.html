160  template<typename T> T read() {
161  T buf_to_return;
162  unsigned char *p_buf_to_return = reinterpret_cast<unsigned char*>(&buf_to_return);
163 
164  if (isBigEndian()) {
165  for (size_t i=0; i<sizeof(buf_to_return); ++i) {
166  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
167  p_buf_to_return[i] = buf[buf_index++];
168  }
169  } else {
170  for (size_t i=0; i<sizeof(buf_to_return); ++i) {
171  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
172  p_buf_to_return[sizeof(buf_to_return)-1-i] = buf[buf_index++];
173  }
174  }
175 
176  return buf_to_return;
177  }
178 
179  template<typename T> void write(T inv) {
180  unsigned char *p_buf_to_send = reinterpret_cast<unsigned char*>(&inv);
181 
182  if (isBigEndian()) {
183  for (size_t i=0; i<sizeof(inv); ++i) {
184  buf += p_buf_to_send[i];
185  }
186  } else {
187  for (size_t i=0; i<sizeof(inv); ++i) {
188  buf += p_buf_to_send[sizeof(inv)-1-i];
189  }
190  }
191  }
