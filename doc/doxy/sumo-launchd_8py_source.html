447 def read_launch_config(conn):
448  """
449  Read (and return) launch configuration from socket
450  """
451 
452  # Get TraCI message length
453  msg_len_buf = ""
454  while len(msg_len_buf) < 4:
455  msg_len_buf += conn.recv(4 - len(msg_len_buf))
456  msg_len = struct.unpack("!i", msg_len_buf)[0] - 4
457 
458  logging.debug("Got TraCI message of length %d" % msg_len)
459 
460  # Get TraCI command length
461  cmd_len_buf = ""
462  cmd_len_buf += conn.recv(1)
463  cmd_len = struct.unpack("!B", cmd_len_buf)[0] - 1
464  if cmd_len == -1:
465  cmd_len_buf = ""
466  while len(cmd_len_buf) < 4:
467  cmd_len_buf += conn.recv(4 - len(cmd_len_buf))
468  cmd_len = struct.unpack("!i", cmd_len_buf)[0] - 5
469 
470  logging.debug("Got TraCI command of length %d" % cmd_len)
471 
472  # Get TraCI command ID
473  cmd_id_buf = ""
474  cmd_id_buf += conn.recv(1)
475  cmd_id = struct.unpack("!B", cmd_id_buf)[0]
476 
477  logging.debug("Got TraCI command 0x%x" % cmd_id)
478 
479  if cmd_id == _CMD_GET_VERSION:
480  # handle get version command
481  handle_get_version(conn)
482  # ...and try reading the launch config again
483  return read_launch_config(conn)
484  elif cmd_id != _CMD_FILE_SEND:
485  raise RuntimeError("Expected CMD_FILE_SEND (0x%x), but got 0x%x" % (_CMD_FILE_SEND, cmd_id))
486 
487  # Get File name
488  fname_len_buf = ""
489  while len(fname_len_buf) < 4:
490  fname_len_buf += conn.recv(4 - len(fname_len_buf))
491  fname_len = struct.unpack("!i", fname_len_buf)[0]
492  fname = conn.recv(fname_len)
493  if fname != "sumo-launchd.launch.xml":
494  raise RuntimeError('Launch configuration must be named "sumo-launchd.launch.xml", got "%s" instead.' % fname)
495 
496  logging.debug('Got CMD_FILE_SEND for "%s"' % fname)
497 
498  # Get File contents
499  data_len_buf = ""
500  while len(data_len_buf) < 4:
501  data_len_buf += conn.recv(4 - len(data_len_buf))
502  data_len = struct.unpack("!i", data_len_buf)[0]
503  data = conn.recv(data_len)
504 
505  logging.debug('Got CMD_FILE_SEND with data "%s"' % data)
506 
507  # Send OK response
508  response = struct.pack("!iBBBi", 4+1+1+1+4, 1+1+1+4, _CMD_FILE_SEND, 0x00, 0x00)
509  conn.send(response)
510 
511  return data
