130 void rbvtrTraCIMobility::handleSelfMessage(cMessage *msg)
131 {
132  if (msg == startAccidentMsg) {
133  commandSetSpeed(0);
134  simtime_t accidentDuration = par("accidentDuration");
135  scheduleAt(simTime() + accidentDuration, stopAccidentMsg);
136  accidentCount--;
137  }
138  else if (msg == stopAccidentMsg) {
139  commandSetSpeed(-1);
140  if (accidentCount > 0) {
141  simtime_t accidentInterval = par("accidentInterval");
142  scheduleAt(simTime() + accidentInterval, startAccidentMsg);
143  }
144  }
145 }

174 void rbvtrTraCIMobility::move()
175 {
176  // ensure we're not called twice in one time step
177  ASSERT(lastUpdate != simTime());
178 
179  // keep statistics (for current step)
180  currentPosXVec.record(lastPosition.x);
181  currentPosYVec.record(lastPosition.y);
182 
183  // keep statistics (relative to last step)
184  if (statistics.startTime != simTime()) {
185  simtime_t updateInterval = simTime() - this->lastUpdate;
186  this->lastUpdate = simTime();
187 
188  double distance = lastPosition.distance(nextPos);
189  statistics.totalDistance += distance;
190  statistics.totalTime += updateInterval;
191  if (speed != -1) {
192  statistics.minSpeed = std::min(statistics.minSpeed, speed);
193  statistics.maxSpeed = std::max(statistics.maxSpeed, speed);
194  currentSpeedVec.record(speed);
195  if (last_speed != -1) {
196  acceleration = (speed - last_speed) / updateInterval;
197  currentAccelerationVec.record(acceleration);
198  }
199  last_speed = speed;
200  } else {
201  last_speed = -1;
202  speed = -1;
203  }
204  }
205 
206  lastPosition = nextPos;
207  if (ev.isGUI()) updateDisplayString();
208  fixIfHostGetsOutside();
209  emitMobilityStateChangedSignal();
210  updateVisualRepresentation();
211 }
